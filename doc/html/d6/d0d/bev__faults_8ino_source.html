128/**
129 * @brief Fault Checking API
130 *
131 * @paragraph CheckFaults Fault Manager Fault Handling
132 * The API checks the HighWord and LowWord individually, then if
133 * the word is evaluated as non-zero it's bits are checked. Once a
134 * raised bit is found, the fault map is used to determine if it's
135 * critical and log a readable message. If the fault is critical the
136 * function returns a bad return code, THIS FUNCTION WILL NOT CHANGE
137 * STATE. If the fault was non-critical, faults are cleared (assuming
138 * no other faults were raised, if so then faults will not be cleared).
139 *
140 *
141 * @return code_t
142 */
143code_t IFaultManager::CheckFaults()
144{
145 Fault_t fault;
146 code_t ret = OK;
147 bool clearFaults = false;
148
149 uint16_t *wordPtr;
150 uint16_t *words[2] = {HighWord, LowWord};
151
152 for (int j=0; j<2; j++){
153 wordPtr = words[j];
154
155 if (*wordPtr)
156 {
157 for (int i=0; i<16; i++)
158 {
159 if (*wordPtr & (0x1 < i))
160 {
161 fault = FaultMap[i + (j*16)];
162 Log.error(fault.string);
163
164 if (fault.critical)
165 ret = FAIL;
166 else
167 clearFaults = true;
168 }
169 }
170
171 }
172 }
173
174 /** Changing the error state will be done by the task */
175 // if (ret != OK)
176 // {
177 // // ChangeState(ERROR_STATE);
178 // return ret;
179 // }
180
181 if (clearFaults && ret == OK)
182 {
183 ClearFaults();
184 }
185
186 return ret;
187}

204/**
205 * @brief PM100DX Clear Faults
206 *
207 * @return code_t
208 *
209 * @paragraph PM100DX ClearFaults PM100DX Clearing Faults
210 * To clear faults, the clear faults command is sent using the param command message.
211 * The Fault Manager is running in it's own task, so it should be noted when trying to
212 * enabled the inverter, that a ClearFaults command is already being sent if no other
213 * critical faults are present. This pertains to "CAN Command Message Lost Fault", which
214 * is raised when a heartbeat message isn't being sent.
215 *
216 */
217void PM100DX_FaultManager::ClearFaults()
218{
219 static unsigned buffer[M193_Read_Write_Param_Command_DLC] = {
220 20, 0, 1, 0, 0, 0, 0, 0
221 };
222
223 SendMessage(M193_Read_Write_Param_Command_CANID, buffer,
224 M193_Read_Write_Param_Command_DLC);
225}

265/**
266 * @brief Fault Manager FreeRTOS Task
267 *
268 * Creates instances of Fault Managers, checks faults at low frequency, high priority.
269 */
270void vFaultManager(__attribute__((unused)) void * pvParameters)
271{
272 static PM100DX_FaultManager rmsPostMgr = PM100DX_FaultManager(
273 &DBCParser.M171_Fault_Codes.D2_Post_Fault_Hi,
274 &DBCParser.M171_Fault_Codes.D1_Post_Fault_Lo,
275 PM100DX_POST_FaultMap);
276
277 static PM100DX_FaultManager rmsRunMgr = PM100DX_FaultManager(
278 &DBCParser.M171_Fault_Codes.D4_Run_Fault_Hi,
279 &DBCParser.M171_Fault_Codes.D3_Run_Fault_Lo,
280 PM100DX_RUN_FaultMap);
281
282 static OrionBMS2_FaultManager bmsDtcMgr = OrionBMS2_FaultManager(
283 &DBCParser.BMS_Faults1.DTCStatus1,
284 &DBCParser.M171_Fault_Codes.D3_Run_Fault_Lo,
285 OrionBMS2_DTCStatus_FaultMap);
286
287 TickType_t xLastWakeTime;
288 const TickType_t xFrequency = pdMS_TO_TICKS(1000);
289
290 // Initialise the xLastWakeTime variable with the current time.
291 xLastWakeTime = xTaskGetTickCount();
292
293 for( ;; )
294 {
295
296 if (rmsPostMgr.CheckFaults() != OK)
297 {
298 ChangeState(ERROR_STATE);
299 }
300
301 if (rmsRunMgr.CheckFaults() != OK)
302 {
303 ChangeState(ERROR_STATE);
304 }
305
306 if (bmsDtcMgr.CheckFaults() != OK)
307 {
308 ChangeState(ERROR_STATE);
309 }
310
311 // Wait for the next cycle.
312 vTaskDelayUntil(&xLastWakeTime, xFrequency);
313 }
