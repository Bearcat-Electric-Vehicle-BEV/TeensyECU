59void CAN2Str(const CAN_message_t &msg, char *buffer, size_t len) {
60
61 snprintf(buffer, len, "MB:%d OVERRUN:%d LEN:%d EXT:%d TS:%05d ID:%04lX BUFFER: ",
62 msg.mb, msg.flags.overrun, msg.len, msg.flags.extended,
63 msg.timestamp, msg.id);
64
65 /* Adding char '0' to numeric returns ascii value */
66 char tmpBuf[msg.len] = {0};
67 for ( uint8_t i = 0; i < msg.len; i++ ) {
68 tmpBuf[i] = msg.buf[i] + '0';
69 }
70
71 /* Append to buffer */
72 strncat(buffer, tmpBuf, msg.len);
73
74}
