43void setup() {
44
45 // Serial.begin(115200);

49 // Set Pin Modes
50 pinMode(PIN_VEHICLE_PWR, INPUT);
51 pinMode(PIN_Brake_Light, OUTPUT);

53 pinMode(PIN_PRECHARGE_FINISHED, INPUT);
54 pinMode(PIN_READY_TO_GO, OUTPUT);

56 pinMode(PIN_CAN_TRANS_STDBY, OUTPUT);
57 pinMode(PIN_SDA, OUTPUT);
58 pinMode(PIN_SCL, OUTPUT);
59 pinMode(PIN_ACCEL_0, INPUT);
60 pinMode(PIN_ACCEL_1, INPUT);
61 pinMode(PIN_BRAKE_POS, INPUT);
62 pinMode(PIN_SPEAKER, OUTPUT);
63 pinMode(PIN_RESET, INPUT);
64 pinMode(PIN_FORWARD_SWITCH, INPUT);
65 pinMode(PIN_FORWARD_ENABLE, OUTPUT);
66
67 /* tranceiver enable pin */
68 digitalWrite(PIN_CAN_TRANS_STDBY, LOW);
69
70 CANInit();
71
72 /**
73 * @paragraph Static Allocation vs Dynamic Allocation
74 * On embedded targets static allocation is typically favored, however with
75 * the high stakes of a stack overflow causing the car to fail. The temptation
76 * to go with heap allocation is real. Most definiently slower, however lowers
77 * probability of stack overflow. With that being said, stack overflow checks
78 * should be in place. There exist tools or methods to determine how much heap
79 * is being used.
80 */
81 xTaskCreate(vStateMachine, "STATE", 1024, nullptr, 10, &pxStateMachineHandle);
82 xTaskCreate(vETCTask, "ETC", 1024, nullptr, 5, &pxETCTaskHandle);
83 xTaskCreate(vFaultManager, "FAULT", 1024, nullptr, 6, &pxFaultManagerHandle);
84
85 /** @note Execution doesn't go beyond scheduler */
86 vTaskStartScheduler();
87
88 for(;;) {}
89
90}
