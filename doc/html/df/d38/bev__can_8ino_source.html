67/**
68 * SendMessage
69 *
70 * @param id CAN message ID
71 * @param buffer Buffer to create message from
72 * @param len Length of Buffer (DLC)
73 *
74 * @return True/Success, False/Failed
75 *
76 * Constructs a CAN msg from parameters and attempts to write to CAN bus. Fails
77 * if the user provides an invalid parameter.
78 *
79 */
80code_t SendMessage(unsigned id, const unsigned *buffer, unsigned len) {
81
82 static CAN_message_t msg;
83
84 if (buffer == nullptr)
85 return FAIL;
86
87 msg.id = id;
88 msg.flags.extended = 0;
89
90 for (unsigned i=0; i<len; i++){
91 msg.buf[i] = buffer[i];
92 }
93
94 Can0.write(msg);
95
96 return OK;
97
98}

123 msg.flags.extended = 0;
124 msg.len = M192_Command_Message_DLC;

126 msg.buf[0] = (cmd->Torque_Command * 10) % 256;

129 /** @todo should speed mode be disabled? */
130 // msg.buf[2] = SpeedCommand % 256;
131 // msg.buf[3] = int(SpeedCommand / 256);
132 msg.buf[2] = msg.buf[3] = 0;
133
134 msg.buf[4] = (cmd->Direction_Command & 0x1);
135
136 msg.buf[5] = (cmd->InvertedEnable & 0x1);
137 msg.buf[5] |= (cmd->Inverter_Discharge & 0x2);
138
139 // msg.buf[5] |= (cmd->Speed_Mode_Enabled & 0x4);
140 msg.buf[5] |= (cmd->RollingCounter & 0x8);
141
142 msg.buf[6] = (cmd->Torque_Limit_Command * 10) % 256;
143 msg.buf[7] = int(cmd->Torque_Limit_Command * (10 / 256));
144
145 Can0.write(msg);

147}
